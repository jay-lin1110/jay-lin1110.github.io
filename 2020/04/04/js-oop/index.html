借用构造函数继承

寄生式组合继承

var Animal = (() => {
  function Animal(sort){
    this.sort = sort
  }
  Animal.prototype.getSort = function(){
    console.log(this.sort)
  }
  Animal.prototype.getFood = function(){
    console.log('any')
  }
  Animal.type = 'Animal'
  Animal.getType = function() {
    console.log(this.type)
  }
  return Animal
})()

var Cat = (() => {
  function Cat(sort, food){
    Animal.call(this, sort)
    this.food = food 
  }
  // 子类继承父类的方法
  Cat.prototype = Object.create(Animal.prototype, {
    constructor: {
      value: Cat
    }
  })
  // 子类重写父类的方法
  Cat.prototype.getFood = function(){
    console.log(this.food)
  }
  return Cat
})()

var meow = new Cat('cat', 'fish')
meow.getSort()
meow.getFood()
console.log(Animal.type)
console.log(Cat.type)// undefined
Animal.getType()
Cat.getType()// Cat.getType is not a function

ES6 的继承

class Cellphone {
  constructor(sort){
    this.sort = sort
  }
  getSort(){
    console.log(this.sort)
  }
  getPrice(){
    console.log(`${this.sort} is ￥0.`)
  }
  // 静态属性
  static type = 'Cellphone'
  // 静态方法
  static getType(){
    console.log(this.type)
  }
}

class IPhone extends Cellphone {
  constructor(sort, price){
    super(sort)
    this.price = price
  }
  getPrice(){
    console.log(`${this.sort} is ￥${this.price}.`)
  }
}

const iphone = new IPhone('IPhone', 7777)
iphone.getSort()
iphone.getPrice()
console.log(IPhone.type)
IPhone.getType()

ES5 VS ES6

直接调用

ES5 的构造函数可以不使用 new 关键字作为函数调用直接执行。
ES6 的 class 必须通过 new 关键字调用，否则抛出异常： TypeError: Class constructor IPhone cannot be invoked without 'new'
